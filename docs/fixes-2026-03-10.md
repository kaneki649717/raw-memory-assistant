// 【新增】提取端口号
const PORT_RE = /\b(?:port|端口)[:\s]*(\d{2,5})\b/gi;

// 【新增】提取参数
const PARAM_RE = /\b(?:maxResults|minScore|temperature|top_p|max_tokens)[:\s=]*([0-9.]+)\b/gi;

export function extractEntities(text) {
  const hits = [
    ...(text.match(FILE_RE) ?? []),
    ...(text.match(MODEL_RE) ?? []),
    ...(text.match(CONFIG_RE) ?? []),
    ...(text.match(ERROR_RE) ?? []),
  ];
  
  // 【新增】提取端口号和参数
  let match;
  const portMatches = [];
  const portRe = /\b(?:port|端口)[:\s]*(\d{2,5})\b/gi;
  while ((match = portRe.exec(text)) !== null) {
    portMatches.push(`port:${match[1]}`);
  }
  
  const paramMatches = [];
  const paramRe = /\b(?:maxResults|minScore|temperature|top_p|max_tokens)[:\s=]*([0-9.]+)\b/gi;
  while ((match = paramRe.exec(text)) !== null) {
    paramMatches.push(match[0]);
  }
  
  return uniq([...hits, ...portMatches, ...paramMatches]).slice(0, 20);
}

### 4. 修复 L0 检索召回问题

**问题**：
- L0 检索阈值过高（隐式为 0），导致召回率低
- 缺少话题关键词加权，相关性排序不准确
- 检索字段不完整，缺少 `timelineKey`

**修复**：

#### 4.1 降低检索阈值

```javascript
export function findL0ByQuery(query, limit = 8) {
  const matches = [...store.items].reverse()
    .map((item) => {
      const haystack = [
        item.summaryShort, 
        item.topic ?? "", 
        item.actionType ?? "", 
        ...(item.entities ?? []), 
        item.resultTag ?? "",
        item.topicKey ?? "",
        item.timelineKey ?? "", // 【新增】
      ].join("\n");
      
      const score = queryMatchScore(query, haystack);
      return { item, score };
    })
    .filter((entry) => entry.score > 0.05) // 【修复】降低阈值
    .sort((a, b) => b.score - a.score)
    .slice(0, limit)
    .map((entry) => entry.item);
  
  // 【修复】如果完全没匹配到，返回最近的几条
  if (matches.length === 0) {
    return store.items.slice(-Math.min(limit, 8)).reverse();
  }
  
  return matches;
}
```

### 5. 优化 L0 创建频率

**问题**：
- `shouldSkipL0Turn` 过滤规则过于严格
- 导致 Event(110) vs L0(64) 比例失衡
- 一些重要对话没有创建 L0 目录

**修复**：

```javascript
function shouldSkipL0Turn(userText, assistantText, actionType) {
  const combined = sanitizeMemoryText(`${userText} ${assistantText}`).toLowerCase();
  const normalizedUser = sanitizeMemoryText(userText).toLowerCase();
  if (!combined) return true;
  
  // 【优化】只跳过真正无意义的对话
  if (["你好", "hi", "hello", "在吗", "ok"].includes(normalizedUser) && combined.length < 30) return true;
  if (combined.includes("消息已发送") && combined.includes("测试消息") && combined.length < 50) return true;
  if (combined.includes("reply with ok") || combined.includes("仅回复ok") || combined.includes("只回ok")) return true;
  
  // 【移除】不再跳过"会话推进"类型的短对话
  return false;
}
```

**影响文件**：
- `runtime/ingest.mjs`

**设计决策**：
- L0 与 Event 不需要 1:1 对应
- L0 是"目录导航层"，只在重要话题切换时创建
- Event 是完整事件日志
- 当前 L0(64) vs Event(110) 约 58% 比例合理

## 技术细节

### 编码修复脚本

创建了 `scripts/fix-encoding.mjs` 用于批量修复编码：

```javascript
function fixEncoding(filePath) {
  let raw = fs.readFileSync(filePath, "utf-8");
  raw = raw.replace(/^\uFEFF/, ""); // 移除 BOM
  const data = JSON.parse(raw);
  
  function cleanObject(obj) {
    if (typeof obj === "string") {
      return obj
        .replace(/�/g, "")
        .replace(/\uFFFD/g, "")
        .replace(/[\x00-\x08\x0B\x0C\x0E-\x1F\x7F]/g, "");
    }
    if (Array.isArray(obj)) {
      return obj.map(cleanObject);
    }
    if (obj && typeof obj === "object") {
      const cleaned = {};
      for (const [key, value] of Object.entries(obj)) {
        cleaned[key] = cleanObject(value);
      }
      return cleaned;
    }
    return obj;
  }
  
  const cleaned = cleanObject(data);
  const json = JSON.stringify(cleaned, null, 2);
  fs.writeFileSync(filePath, json, { encoding: "utf-8" });
}
```
